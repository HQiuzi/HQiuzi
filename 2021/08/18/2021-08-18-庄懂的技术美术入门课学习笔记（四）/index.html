            #pragma multi_compile_fog
            #pragma target 3.0

			uniform float3 _BaseCol;
			//uniform float3 _LightCol;
			uniform float _SpecularPow;
			uniform float _Intensity;
			uniform float _EnvIntensity;
			uniform sampler2D _Occlusion;
			uniform float3 _EnvUpCol;
			uniform float3 _EnvDownCol;
			uniform float3 _EnvSideCol;

			uniform sampler2D _NormalMap;
			uniform samplerCUBE _CubeMap;
			uniform float _CubemapMip;
			uniform float _FresnelExp;
			uniform float _EnvSpecInt;

            struct VertexInput {
                float4 vertex : POSITION;
				float3 normal : NORMAL;//è·å–æ³•çº¿ä¿¡æ¯
				float2 uv0 : TEXCOORD0;//å°†æ¨¡å‹uvä¿¡æ¯è¾“å…¥è¿›æ¥
				float4 tangent : TANGENT;//åˆ‡çº¿
            };
            struct VertexOutput {
                float4 pos : SV_POSITION;
				float2 uv0 : TEXCOORD0;
				float3 nDirWS : TEXCOORD1;
				float3 tDirWS : TEXCOORD2;
				float3 bDirWS : TEXCOORD3;
				float4 posWS : TEXCOORD4;//ä¸–ç•Œå®šç‚¹ä¿¡æ¯
				LIGHTING_COORDS(5,6)
				
            };

			//è¾“å…¥ç»“æ„>>>é¡¶ç‚¹Shader>>>è¾“å‡ºç»“æ„
            VertexOutput vert (VertexInput v) {
                VertexOutput o = (VertexOutput)0;
                o.pos = UnityObjectToClipPos( v.vertex );//è½¬æ¢åˆ°è£å‰ªç©ºé—´
				o.posWS = mul(unity_ObjectToWorld,v.vertex);
				o.nDirWS = UnityObjectToWorldNormal(v.normal);//ç”±æ¨¡å‹æ³•çº¿ä¿¡æ¯æ¢ç®—çš„ä¸–ç•Œç©ºé—´æ³•çº¿ä¿¡æ¯
				o.tDirWS = normalize(mul(unity_ObjectToWorld,float4(v.tangent.xyz,0.0)).xyz);
				o.bDirWS = normalize(cross(o.nDirWS,o.tDirWS)*v.tangent.w);
				o.uv0 = v.uv0;

				TRANSFER_VERTEX_TO_FRAGMENT(o)

                return o;
            }
            float4 frag(VertexOutput i) : COLOR {
			//å‡†å¤‡å‘é‡ 

				float3 lDir = _WorldSpaceLightPos0.xyz;//Pos0æ˜¯å¹³è¡Œå…‰ï¼Œxyzä»£è¡¨æ–¹å‘åæ ‡ï¼Œè¿™é‡Œæœ€å¥½åšä¸€ä¸‹æ­£åˆ™åŒ–
				//å‘é‡å‡†å¤‡
				float3 vDirWS = normalize(_WorldSpaceCameraPos.xyz - i.posWS.xyz);
				float3 nDirTS = UnpackNormal(tex2D(_NormalMap,i.uv0)).rgb;
				float3x3 TBN = float3x3(i.tDirWS,i.bDirWS,i.nDirWS);
				float3 nDirWS = normalize(mul(nDirTS,TBN));//tangent to world
				float3 vrDirWS = reflect(-vDirWS,nDirWS);
				float3 rDirWS = reflect(-lDir,nDirWS);

				float shadow = LIGHT_ATTENUATION(i);

////// Light:
				float nDotl = dot(nDirWS,lDir);
				float vDotr = dot(vDirWS,rDirWS);
				float vDotn = dot(vDirWS,nDirWS);

				float lambert = max(0.0,nDotl);
				//float specCol = var_SpecTex.rgb;
				//float specPow = lerp(_SpecularPow,1.0,var_SpecTex.a);  //é«˜å…‰è´´å›¾,ç”¨rgbé€šé“å­˜å‚¨é¢œè‰²ï¼Œaé€šé“å­˜å‚¨å…‰æ»‘åº¦
				float phong = pow(max(0.0,vDotr),_SpecularPow);

				///ç›´æ¥å…‰ç…§
				float3 lightRes =_LightColor0*(_BaseCol*lambert+phong)*_Intensity*shadow;

////// env:

				//ç¯å¢ƒå…‰æ¼«åå°„
				float upMask = max(0.0,nDirWS.g);
				float downMask = max(0.0,-nDirWS.g);
				float sideMask = 1.0-upMask-downMask;
				float3 envCol= _EnvUpCol*upMask+_EnvDownCol*downMask+_EnvSideCol*sideMask;

				//ç¯å¢ƒå…‰é•œé¢åå°„
				//float cubemapMip=lerp(_CubemapMip,1.0,var_SpecTex.a);  //é«˜å…‰è´´å›¾
				float3 cubemap = texCUBElod(_CubeMap, float4(vrDirWS,_CubemapMip));
				float fresnel = pow(1.0-vDotn,_FresnelExp);

				//ç¯å¢ƒå…‰é®ç½©
				float occlusion=tex2D(_Occlusion,i.uv0);

				//ç¯å¢ƒå…‰
				float3 envSpec = cubemap*_EnvSpecInt*fresnel;
				float3 envDiff = envCol*_BaseCol*_EnvIntensity;
				float3 envRes = (envSpec+envDiff)*occlusion;

				float3 finalRGB = lightRes+envRes;

                return float4(finalRGB,1.0);//æ³¨æ„ä»£ç è§„èŒƒï¼Œè½¬åŒ–ä¸ºfloat4 rgba
            }
            ENDCG
        }
